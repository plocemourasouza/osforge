Zero dependências externas · Python built-in sqlite3 · FTS5 full-text search · banco global (~/.osforge/) ou local (.osforge/)

Comandos essenciais
# INTAKE — retomada em ~50 tokens
osforge-db resume projeto
# → fase=spec-builder | resume=...

# TRACK — progressão
osforge-db set-phase proj fase complete
osforge-db add-decision proj "..." --category=arch
osforge-db set-resume proj "próximo passo"

# SEARCH — FTS5 cross-project
osforge-db search "Prisma RLS multi-tenant"

# MIGRAÇÃO de status.yaml
osforge-db import-yaml .osforge/status.yaml slug
